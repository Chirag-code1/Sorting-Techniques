//**************SELECTION SORT********************



#include <stdio.h>

int main() {
   int n=0, i=0, j=0, temp=0;
   scanf("%d",&n);
   int arr[n];
   for(i=0;i<n;i++){
       scanf("%d",&arr[i]);
   }
   for(i=0;i<n-1;i++){
       for(j=i+1;j<n;j++){
           if(arr[i]>arr[j]){
               temp=arr[j];
               arr[j]=arr[i];
               arr[i]=temp;
           }
       }
   }
   for(i=0;i<n;i++){
   printf("%d  ",arr[i]);
}
    
    return 0;
}
